### 5. 意图路由配置化

```typescript
// src/services/intentRouter.ts

interface IntentRoute {
  name: string;
  keywords: string[];
  responseTemplate: string;
  citations: string[];  // citation IDs
}

const INTENT_ROUTES: IntentRoute[] = [
  {
    name: "考勤制度",
    keywords: ["工作", "考勤", "年假", "假期", "休假"],
    responseTemplate: "...",
    citations: ["cit-3"],
  },
  {
    name: "系统架构",
    keywords: ["架构", "设计", "安全", "向量", "embedding"],
    responseTemplate: "...",
    citations: ["cit-1", "cit-2"],
  },
  // ...
];

export function matchIntent(query: string): IntentRoute | null {
  const lower = query.toLowerCase();
  return INTENT_ROUTES.find(route =>
    route.keywords.some(kw => lower.includes(kw))
  ) || null;
}
```

> P0 保留此 Mock 路由用于前端原型验证；P1 接入真实后端后，意图识别由 `chat_service.py` 的 progress 事件推送。

## 前后端类型对齐

前端 `src/types/api.ts` 直接对齐契约文档（02-接口契约文档.md），与后端 Pydantic schemas 一一对应：

| 前端类型 | 后端 Pydantic | 说明 |
|---------|-------------|------|
| `KbDocument` | `DocumentResponse` | 对齐 |
| `RefChunk` | `RefChunkResponse` | 对齐 |
| `ChatRequest` | `ChatRequest` | { question, session_id } |
| `SSEEvent` | (SSE 协议) | progress / token / done / error |
| `KbConfig` | `ConfigResponse` | 对齐 |
| `Session` | `SessionResponse` | 对齐 |
| `ApiResponse<T>` | `CommonResponse` | { code, data } |
